"""
Напишите программу, которая запрашивает у пользователя Имя, возраст, рост и вес
а затем выводить приветствие с указанием имени и возраста и индекс массы тела.

Индекс массы тела вычисляется как вес в кг делённый на рост в метрах возведённый в квадрат.
"""

try:
    name, age, height, weight = input("Как Вас зовут? "), int(input("А сколько Вам лет? ")), float(input("Какой у Вас рост? ")), float(input("Какой у Вас вес? "))
    
    index_weight = weight // (height ** 2) 

    print(f"Привет, {name}. Тебе {age} лет!\nТвой индекс массы тела -> равна => {index_weight}")
    
except:
    print("Вы ввели чтото не понятное!")
